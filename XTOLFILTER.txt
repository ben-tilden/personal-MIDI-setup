# xtouch to launchkey filter

# filter messages that don't have 11 or C1 as their routing (CC)

IF M1 > C1
  BLOCK
  EXIT
END

# allow for drumParams to pass
IF M0 == B0
  IF M1 < $14
    BLOCK
    EXIT
  END
  IF M1 > $31
    IF M1 < $46
      BLOCK
      EXIT
    END
  END
  IF M1 > $48
    IF M1 < $53
      BLOCK
      EXIT
    END
  END
  IF M1 > $53
    IF M1 < $116
      BLOCK
      EXIT
    END
  END
  IF M1 > $119
    BLOCK
    EXIT
  END
END

N0-F = XX +B
